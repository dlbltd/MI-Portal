Go to https://workers.cloudflare.com and sign up (free) // 2. Click "Create a Worker" // 3. Delete all the default code // 4. Paste this entire file in // 5. Click "Save and Deploy" // 6. Copy the worker URL (e.g. https://dlb-proxy.YOUR-NAME.workers.dev) // 7. Paste that URL into the "Proxy URL" field in trackops-sync.html // ============================================================ const ALLOWED_ORIGIN = 'https://dlbltd.github.io'; // Your GitHub Pages URL const TRACKOPS_BASE = 'https://dlbinvestigations.viewcases.com/api/v1'; export default { async fetch(request) { // Handle CORS preflight if (request.method === 'OPTIONS') { return new Response(null, { status: 204, headers: corsHeaders() }); } // Only allow requests from your GitHub Pages site const origin = request.headers.get('Origin') || ''; if (origin !== ALLOWED_ORIGIN) { return new Response('Forbidden', { status: 403 }); } // Build the TrackOps URL from the incoming path const url = new URL(request.url); const apiPath = url.searchParams.get('path') || '/cases'; const fullUrl = TRACKOPS_BASE + apiPath; // Forward the request to TrackOps, passing auth headers through const trackopsReq = new Request(fullUrl, { method: request.method, headers: { 'Authorization': request.headers.get('Authorization') || '', 'Content-Type': 'application/json', 'Accept': 'application/json', }, body: ['GET', 'HEAD'].includes(request.method) ? null : request.body, }); try { const response = await fetch(trackopsReq); const body = await response.text(); return new Response(body, { status: response.status, headers: { 'Content-Type': 'application/json', ...corsHeaders() } }); } catch (err) { return new Response(JSON.stringify({ error: err.message }), { status: 502, headers: { 'Content-Type': 'application/json', ...corsHeaders() } }); } } }; function corsHeaders() { return { 'Access-Control-Allow-Origin': ALLOWED_ORIGIN, 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'Authorization, Content-Type, Accept', 'Access-Control-Max-Age': '86400', }; }
